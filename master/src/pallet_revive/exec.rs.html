80/// Combined key type for both fixed and variable sized storage keys.
81#[derive(Debug)]
82pub enum Key {
83	/// Variant for fixed sized keys.
84	Fix([u8; 32]),
85	/// Variant for variable sized keys.
86	Var(VarSizedKey),
87}
88
89impl Key {
90	/// Reference to the raw unhashed key.
91	pub fn unhashed(&self) -> &[u8] {
92		match self {
93			Key::Fix(v) => v.as_ref(),
94			Key::Var(v) => v.as_ref(),
95		}
96	}
97
98	/// The hashed key that has be used as actual key to the storage trie.
99	pub fn hash(&self) -> Vec<u8> {
100		match self {
101			Key::Fix(v) => blake2_256(v.as_slice()).to_vec(),
102			Key::Var(v) => Blake2_128Concat::hash(v.as_slice()),
103		}
104	}
105
106	pub fn from_fixed(v: [u8; 32]) -> Self {
107		Self::Fix(v)
108	}
109
110	pub fn try_from_var(v: Vec<u8>) -> Result<Self, ()> {
111		VarSizedKey::try_from(v).map(Self::Var).map_err(|_| ())
112	}
113}

146/// The type of origins supported by the revive pallet.
147#[derive(Clone, Encode, Decode, PartialEq, TypeInfo, RuntimeDebugNoBound)]
148pub enum Origin<T: Config> {
149	Root,
150	Signed(T::AccountId),
151}
152
153impl<T: Config> Origin<T> {
154	/// Creates a new Signed Caller from an AccountId.
155	pub fn from_account_id(account_id: T::AccountId) -> Self {
156		Origin::Signed(account_id)
157	}
158	/// Creates a new Origin from a `RuntimeOrigin`.
159	pub fn from_runtime_origin(o: OriginFor<T>) -> Result<Self, DispatchError> {
160		match o.into() {
161			Ok(RawOrigin::Root) => Ok(Self::Root),
162			Ok(RawOrigin::Signed(t)) => Ok(Self::Signed(t)),
163			_ => Err(BadOrigin.into()),
164		}
165	}
166	/// Returns the AccountId of a Signed Origin or an error if the origin is Root.
167	pub fn account_id(&self) -> Result<&T::AccountId, DispatchError> {
168		match self {
169			Origin::Signed(id) => Ok(id),
170			Origin::Root => Err(DispatchError::RootNotAllowed),
171		}
172	}
173
174	/// Make sure that this origin is mapped.
175	///
176	/// We require an origin to be mapped in order to be used in a `Stack`. Otherwise
177	/// [`Stack::caller`] returns an address that can't be reverted to the original address.
178	fn ensure_mapped(&self) -> DispatchResult {
179		match self {
180			Self::Root => Ok(()),
181			Self::Signed(account_id) if T::AddressMapper::is_mapped(account_id) => Ok(()),
182			Self::Signed(_) => Err(<Error<T>>::AccountUnmapped.into()),
183		}
184	}

575/// When calling an address it can either lead to execution of contract code or a pre-compile.
576enum ExecutableOrPrecompile<T: Config, E: Executable<T>, Env> {
577	/// Contract code.
578	Executable(E),
579	/// Code inside the runtime (so called pre-compile).
580	Precompile { instance: PrecompileInstance<Env>, _phantom: PhantomData<T> },
581}
582
583impl<T: Config, E: Executable<T>, Env> ExecutableOrPrecompile<T, E, Env> {
584	fn as_executable(&self) -> Option<&E> {
585		if let Self::Executable(executable) = self {
586			Some(executable)
587		} else {
588			None
589		}
590	}
591
592	fn is_pvm(&self) -> bool {
593		match self {
594			Self::Executable(e) => e.code_info().is_pvm(),
595			_ => false,
596		}
597	}
598
599	fn as_precompile(&self) -> Option<&PrecompileInstance<Env>> {
600		if let Self::Precompile { instance, .. } = self {
601			Some(instance)
602		} else {
603			None
604		}
605	}
606
607	#[cfg(any(feature = "runtime-benchmarks", test))]
608	fn into_executable(self) -> Option<E> {
609		if let Self::Executable(executable) = self {
610			Some(executable)
611		} else {
612			None
613		}
614	}
615}

626		/// This frame was created by `seal_delegate_call` and hence uses different code than
627		/// what is stored at [`Self::Call::dest`]. Its caller ([`DelegatedCall::caller`]) is the
628		/// account which called the caller contract
629		delegated_call: Option<DelegateInfo<T>>,
630	},
631	Instantiate {
632		/// The contract or signed origin which instantiates the new contract.
633		sender: T::AccountId,
634		/// The executable whose `deploy` function is run.
635		executable: E,
636		/// A salt used in the contract address derivation of the new contract.
637		salt: Option<&'a [u8; 32]>,
638		/// The input data is used in the contract address derivation of the new contract.
639		input_data: &'a [u8],
640	},
641}

643/// Describes the different states of a contract as contained in a `Frame`.
644enum CachedContract<T: Config> {
645	/// The cached contract is up to date with the in-storage value.
646	Cached(ContractInfo<T>),
647	/// A recursive call into the same contract did write to the contract info.
648	///
649	/// In this case the cached contract is stale and needs to be reloaded from storage.
650	Invalidated,
651	/// The current contract executed `terminate` and removed the contract.
652	///
653	/// In this case a reload is neither allowed nor possible. Please note that recursive
654	/// calls cannot remove a contract as this is checked and denied.
655	Terminated,
656	/// The frame is associated with pre-compile that has no contract info.
657	None,
658}
659
660impl<T: Config> Frame<T> {
661	/// Return the `contract_info` of the current contract.
662	fn contract_info(&mut self) -> &mut ContractInfo<T> {
663		self.contract_info.get(&self.account_id)
664	}
665
666	/// Terminate and return the `contract_info` of the current contract.
667	///
668	/// # Note
669	///
670	/// Under no circumstances the contract is allowed to access the `contract_info` after
671	/// a call to this function. This would constitute a programming error in the exec module.
672	fn terminate(&mut self) -> ContractInfo<T> {
673		self.contract_info.terminate(&self.account_id)
674	}
675}
676
677/// Extract the contract info after loading it from storage.
678///
679/// This assumes that `load` was executed before calling this macro.
680macro_rules! get_cached_or_panic_after_load {
681	($c:expr) => {{
682		if let CachedContract::Cached(contract) = $c {
683			contract
684		} else {
685			panic!(
686				"It is impossible to remove a contract that is on the call stack;\
687				See implementations of terminate;\
688				Therefore fetching a contract will never fail while using an account id
689				that is currently active on the call stack;\
690				qed"
691			);
692		}
693	}};
694}
695
696/// Same as [`Stack::top_frame`].
697///
698/// We need this access as a macro because sometimes hiding the lifetimes behind
699/// a function won't work out.
700macro_rules! top_frame {
701	($stack:expr) => {
702		$stack.frames.last().unwrap_or(&$stack.first_frame)
703	};
704}
705
706/// Same as [`Stack::top_frame_mut`].
707///
708/// We need this access as a macro because sometimes hiding the lifetimes behind
709/// a function won't work out.
710macro_rules! top_frame_mut {
711	($stack:expr) => {
712		$stack.frames.last_mut().unwrap_or(&mut $stack.first_frame)
713	};
714}
715
716impl<T: Config> CachedContract<T> {
717	/// Return `Some(ContractInfo)` if the contract is in cached state. `None` otherwise.
718	fn into_contract(self) -> Option<ContractInfo<T>> {
719		if let CachedContract::Cached(contract) = self {
720			Some(contract)
721		} else {
722			None
723		}
724	}
725
726	/// Return `Some(&mut ContractInfo)` if the contract is in cached state. `None` otherwise.
727	fn as_contract(&mut self) -> Option<&mut ContractInfo<T>> {
728		if let CachedContract::Cached(contract) = self {
729			Some(contract)
730		} else {
731			None
732		}
733	}
734
735	/// Load the `contract_info` from storage if necessary.
736	fn load(&mut self, account_id: &T::AccountId) {
737		if let CachedContract::Invalidated = self {
738			if let Some(contract) =
739				AccountInfo::<T>::load_contract(&T::AddressMapper::to_address(account_id))
740			{
741				*self = CachedContract::Cached(contract);
742			}
743		}
744	}
745
746	/// Return the cached contract_info.
747	fn get(&mut self, account_id: &T::AccountId) -> &mut ContractInfo<T> {
748		self.load(account_id);
749		get_cached_or_panic_after_load!(self)
750	}
751
752	/// Terminate and return the contract info.
753	fn terminate(&mut self, account_id: &T::AccountId) -> ContractInfo<T> {
754		self.load(account_id);
755		get_cached_or_panic_after_load!(mem::replace(self, Self::Terminated))
756	}
757
758	/// Set the status to invalidate if is cached.
759	fn invalidate(&mut self) {
760		if matches!(self, CachedContract::Cached(_)) {
761			*self = CachedContract::Invalidated;
762		}
763	}
764}

775	///
776	/// # Return Value
777	///
778	/// Result<(ExecReturnValue, CodeSize), (ExecError, CodeSize)>

780		origin: Origin<T>,
781		dest: H160,
782		gas_meter: &mut GasMeter<T>,
783		storage_meter: &mut storage::meter::Meter<T>,
784		value: U256,
785		input_data: Vec<u8>,
786		skip_transfer: bool,
787	) -> ExecResult {
788		let dest = T::AddressMapper::to_account_id(&dest);
789		if let Some((mut stack, executable)) = Stack::<'_, T, E>::new(
790			FrameArgs::Call { dest: dest.clone(), cached_info: None, delegated_call: None },
791			origin.clone(),
792			gas_meter,
793			storage_meter,
794			value,
795			skip_transfer,

797			stack

799				.map(|_| stack.first_frame.last_frame_output)
800		} else {
801			if_tracing(|t| {
802				t.enter_child_span(
803					origin.account_id().map(T::AddressMapper::to_address).unwrap_or_default(),
804					T::AddressMapper::to_address(&dest),
805					false,
806					false,
807					value,
808					&input_data,
809					Weight::zero(),
810				);
811			});
812

814
815			if_tracing(|t| match result {
816				Ok(ref output) => t.exit_child_span(&output, Weight::zero()),
817				Err(e) => t.exit_child_span_with_error(e.error.into(), Weight::zero()),
818			});

821		}

935	/// Construct a new frame.
936	///
937	/// This does not take `self` because when constructing the first frame `self` is
938	/// not initialized, yet.
939	fn new_frame<S: storage::meter::State + Default + Debug>(
940		frame_args: FrameArgs<T, E>,
941		value_transferred: U256,
942		gas_meter: &mut GasMeter<T>,
943		gas_limit: Weight,
944		storage_meter: &mut storage::meter::GenericMeter<T, S>,
945		deposit_limit: BalanceOf<T>,
946		read_only: bool,
947		origin_is_caller: bool,
948	) -> Result<Option<(Frame<T>, ExecutableOrPrecompile<T, E, Self>)>, ExecError> {
949		let (account_id, contract_info, executable, delegate, entry_point) = match frame_args {
950			FrameArgs::Call { dest, cached_info, delegated_call } => {
951				let address = T::AddressMapper::to_address(&dest);
952				let precompile = <AllPrecompiles<T>>::get(address.as_fixed_bytes());
953
954				// which contract info to load is unaffected by the fact if this
955				// is a delegate call or not
956				let mut contract = match (cached_info, &precompile) {
957					(Some(info), _) => CachedContract::Cached(info),
958					(None, None) =>
959						if let Some(info) = AccountInfo::<T>::load_contract(&address) {
960							CachedContract::Cached(info)
961						} else {
962							return Ok(None);
963						},
964					(None, Some(precompile)) if precompile.has_contract_info() => {
965						if let Some(info) = AccountInfo::<T>::load_contract(&address) {
966							CachedContract::Cached(info)
967						} else {
968							let info = ContractInfo::new(&address, 0u32.into(), H256::zero())?;
969							CachedContract::Cached(info)
970						}
971					},
972					(None, Some(_)) => CachedContract::None,
973				};
974
975				// in case of delegate the executable is not the one at `address`
976				let executable = if let Some(delegated_call) = &delegated_call {
977					if let Some(precompile) =
978						<AllPrecompiles<T>>::get(delegated_call.callee.as_fixed_bytes())
979					{
980						ExecutableOrPrecompile::Precompile {
981							instance: precompile,
982							_phantom: Default::default(),
983						}
984					} else {
985						let Some(info) = AccountInfo::<T>::load_contract(&delegated_call.callee)
986						else {
987							return Ok(None);
988						};
989						let executable = E::from_storage(info.code_hash, gas_meter)?;
990						ExecutableOrPrecompile::Executable(executable)
991					}
992				} else {
993					if let Some(precompile) = precompile {
994						ExecutableOrPrecompile::Precompile {
995							instance: precompile,
996							_phantom: Default::default(),
997						}
998					} else {
999						let executable = E::from_storage(
1000							contract
1001								.as_contract()
1002								.expect("When not a precompile the contract was loaded above; qed")
1003								.code_hash,
1004							gas_meter,
1005						)?;
1006						ExecutableOrPrecompile::Executable(executable)
1007					}
1008				};
1009
1010				(dest, contract, executable, delegated_call, ExportedFunction::Call)
1011			},
1012			FrameArgs::Instantiate { sender, executable, salt, input_data } => {
1013				let deployer = T::AddressMapper::to_address(&sender);
1014				let account_nonce = <System<T>>::account_nonce(&sender);
1015				let address = if let Some(salt) = salt {
1016					address::create2(&deployer, executable.code(), input_data, salt)
1017				} else {
1018					use sp_runtime::Saturating;
1019					address::create1(
1020						&deployer,
1021						// the Nonce from the origin has been incremented pre-dispatch, so we
1022						// need to subtract 1 to get the nonce at the time of the call.
1023						if origin_is_caller {
1024							account_nonce.saturating_sub(1u32.into()).saturated_into()
1025						} else {
1026							account_nonce.saturated_into()
1027						},
1028					)
1029				};
1030				let contract = ContractInfo::new(
1031					&address,
1032					<System<T>>::account_nonce(&sender),
1033					*executable.code_hash(),
1034				)?;
1035				(
1036					T::AddressMapper::to_fallback_account_id(&address),
1037					CachedContract::Cached(contract),
1038					ExecutableOrPrecompile::Executable(executable),
1039					None,
1040					ExportedFunction::Constructor,
1041				)
1042			},
1043		};
1044
1045		let frame = Frame {
1046			delegate,
1047			value_transferred,
1048			contract_info,
1049			account_id,
1050			entry_point,
1051			nested_gas: gas_meter.nested(gas_limit),
1052			nested_storage: storage_meter.nested(deposit_limit),
1053			allows_reentry: true,
1054			read_only,
1055			last_frame_output: Default::default(),
1056		};
1057
1058		Ok(Some((frame, executable)))
1059	}

1061	/// Create a subsequent nested frame.
1062	fn push_frame(
1063		&mut self,
1064		frame_args: FrameArgs<T, E>,
1065		value_transferred: U256,
1066		gas_limit: Weight,
1067		deposit_limit: BalanceOf<T>,
1068		read_only: bool,
1069	) -> Result<Option<ExecutableOrPrecompile<T, E, Self>>, ExecError> {
1070		if self.frames.len() as u32 == limits::CALL_STACK_DEPTH {
1071			return Err(Error::<T>::MaxCallDepthReached.into());
1072		}
1073
1074		// We need to make sure that changes made to the contract info are not discarded.
1075		// See the `in_memory_changes_not_discarded` test for more information.
1076		// We do not store on instantiate because we do not allow to call into a contract
1077		// from its own constructor.
1078		let frame = self.top_frame();
1079		if let (CachedContract::Cached(contract), ExportedFunction::Call) =
1080			(&frame.contract_info, frame.entry_point)
1081		{
1082			AccountInfo::<T>::insert_contract(
1083				&T::AddressMapper::to_address(&frame.account_id),
1084				contract.clone(),
1085			);
1086		}
1087
1088		let frame = top_frame_mut!(self);
1089		let nested_gas = &mut frame.nested_gas;
1090		let nested_storage = &mut frame.nested_storage;
1091		if let Some((frame, executable)) = Self::new_frame(
1092			frame_args,
1093			value_transferred,
1094			nested_gas,
1095			gas_limit,
1096			nested_storage,
1097			deposit_limit,
1098			read_only,
1099			false,
1100		)? {
1101			self.frames.try_push(frame).map_err(|_| Error::<T>::MaxCallDepthReached)?;
1102			Ok(Some(executable))
1103		} else {
1104			Ok(None)
1105		}
1106	}

1108	/// Run the current (top) frame.
1109	///
1110	/// This can be either a call or an instantiate.
1111	fn run(
1112		&mut self,
1113		executable: ExecutableOrPrecompile<T, E, Self>,
1114		input_data: Vec<u8>,
1115		bump_nonce: BumpNonce,
1116	) -> Result<(), ExecError> {
1117		let frame = self.top_frame();
1118		let entry_point = frame.entry_point;
1119		let is_pvm = executable.is_pvm();
1120
1121		if_tracing(|tracer| {
1122			tracer.enter_child_span(
1123				self.caller().account_id().map(T::AddressMapper::to_address).unwrap_or_default(),
1124				T::AddressMapper::to_address(&frame.account_id),
1125				frame.delegate.is_some(),
1126				frame.read_only,
1127				frame.value_transferred,
1128				&input_data,
1129				frame.nested_gas.gas_left(),
1130			);
1131		});
1132
1133		// The output of the caller frame will be replaced by the output of this run.
1134		// It is also not accessible from nested frames.
1135		// Hence we drop it early to save the memory.
1136		let frames_len = self.frames.len();
1137		if let Some(caller_frame) = match frames_len {
1138			0 => None,
1139			1 => Some(&mut self.first_frame.last_frame_output),
1140			_ => self.frames.get_mut(frames_len - 2).map(|frame| &mut frame.last_frame_output),
1141		} {
1142			*caller_frame = Default::default();
1143		}
1144
1145		self.transient_storage.start_transaction();
1146
1147		let do_transaction = || -> ExecResult {
1148			let caller = self.caller();
1149			let skip_transfer = self.skip_transfer;
1150			let frame = top_frame_mut!(self);
1151			let account_id = &frame.account_id.clone();
1152
1153			if u32::try_from(input_data.len())
1154				.map(|len| len > limits::CALLDATA_BYTES)
1155				.unwrap_or(true)
1156			{
1157				Err(<Error<T>>::CallDataTooLarge)?;
1158			}
1159
1160			// We need to make sure that the contract's account exists before calling its
1161			// constructor.
1162			if entry_point == ExportedFunction::Constructor {
1163				// Root origin can't be used to instantiate a contract, so it is safe to assume that
1164				// if we reached this point the origin has an associated account.
1165				let origin = &self.origin.account_id()?;
1166
1167				let ed = <Contracts<T>>::min_balance();
1168				frame.nested_storage.record_charge(&StorageDeposit::Charge(ed))?;
1169				if self.skip_transfer {
1170					T::Currency::set_balance(account_id, ed);
1171				} else {
1172					T::Currency::transfer(origin, account_id, ed, Preservation::Preserve)
1173						.map_err(|_| <Error<T>>::StorageDepositNotEnoughFunds)?;
1174				}
1175
1176				// A consumer is added at account creation and removed it on termination, otherwise
1177				// the runtime could remove the account. As long as a contract exists its
1178				// account must exist. With the consumer, a correct runtime cannot remove the
1179				// account.
1180				<System<T>>::inc_consumers(account_id)?;
1181
1182				// Contracts nonce starts at 1
1183				<System<T>>::inc_account_nonce(account_id);
1184
1185				if matches!(bump_nonce, BumpNonce::Yes) {
1186					// Needs to be incremented before calling into the code so that it is visible
1187					// in case of recursion.
1188					<System<T>>::inc_account_nonce(caller.account_id()?);
1189				}
1190				// The incremented refcount should be visible to the constructor.
1191				if is_pvm {
1192					<CodeInfo<T>>::increment_refcount(
1193						*executable
1194							.as_executable()
1195							.expect("Precompiles cannot be instantiated; qed")
1196							.code_hash(),
1197					)?;
1198				}
1199			}
1200
1201			// Every non delegate call or instantiate also optionally transfers the balance.
1202			// If it is a delegate call, then we've already transferred tokens in the
1203			// last non-delegate frame.
1204			if frame.delegate.is_none() {
1205				Self::transfer_from_origin(
1206					&self.origin,
1207					&caller,
1208					account_id,
1209					frame.value_transferred,
1210					&mut frame.nested_storage,
1211				)?;
1212			}
1213
1214			// We need to make sure that the pre-compiles contract exist before executing it.
1215			// A few more conditionals:
1216			// 	- Only contracts with extended API (has_contract_info) are guaranteed to have an
1217			//    account.
1218			//  - Only when not delegate calling we are executing in the context of the pre-compile.
1219			//    Pre-compiles itself cannot delegate call.
1220			if let Some(precompile) = executable.as_precompile() {
1221				if precompile.has_contract_info() &&
1222					frame.delegate.is_none() &&
1223					!<System<T>>::account_exists(account_id)
1224				{
1225					// prefix matching pre-compiles cannot have a contract info
1226					// hence we only mint once per pre-compile
1227					T::Currency::mint_into(account_id, T::Currency::minimum_balance())?;
1228					// make sure the pre-compile does not destroy its account by accident
1229					<System<T>>::inc_consumers(account_id)?;
1230				}
1231			}
1232
1233			let mut code_deposit = executable
1234				.as_executable()
1235				.map(|exec| exec.code_info().deposit())
1236				.unwrap_or_default();
1237
1238			let mut output = match executable {
1239				ExecutableOrPrecompile::Executable(executable) =>
1240					executable.execute(self, entry_point, input_data),
1241				ExecutableOrPrecompile::Precompile { instance, .. } =>
1242					instance.call(input_data, self),
1243			}
1244			.and_then(|output| {
1245				if u32::try_from(output.data.len())
1246					.map(|len| len > limits::CALLDATA_BYTES)
1247					.unwrap_or(true)
1248				{
1249					Err(<Error<T>>::ReturnDataTooLarge)?;
1250				}
1251				Ok(output)
1252			})
1253			.map_err(|e| ExecError { error: e.error, origin: ErrorOrigin::Callee })?;
1254
1255			// Avoid useless work that would be reverted anyways.
1256			if output.did_revert() {
1257				return Ok(output);
1258			}
1259
1260			let frame = self.top_frame_mut();
1261
1262			// The deposit we charge for a contract depends on the size of the immutable data.
1263			// Hence we need to delay charging the base deposit after execution.
1264			if entry_point == ExportedFunction::Constructor {
1265				let contract_info = frame.contract_info();
1266				// if we are dealing with EVM bytecode
1267				// We upload the new runtime code, and update the code
1268				if !is_pvm {
1269					// Only keep return data for tracing
1270					let data = if crate::tracing::if_tracing(|_| {}).is_none() {
1271						core::mem::replace(&mut output.data, Default::default())
1272					} else {
1273						output.data.clone()
1274					};
1275
1276					let mut module = crate::ContractBlob::<T>::from_evm_runtime_code(
1277						data,
1278						caller.account_id()?.clone(),
1279					)?;
1280					module.store_code(skip_transfer)?;
1281					code_deposit = module.code_info().deposit();
1282					contract_info.code_hash = *module.code_hash();
1283
1284					<CodeInfo<T>>::increment_refcount(contract_info.code_hash)?;
1285				}
1286
1287				let deposit = contract_info.update_base_deposit(code_deposit);
1288				frame
1289					.nested_storage
1290					.charge_deposit(frame.account_id.clone(), StorageDeposit::Charge(deposit));
1291			}
1292
1293			// The storage deposit is only charged at the end of every call stack.
1294			// To make sure that no sub call uses more than it is allowed to,
1295			// the limit is manually enforced here.
1296			let contract = frame.contract_info.as_contract();
1297			frame
1298				.nested_storage
1299				.enforce_limit(contract)
1300				.map_err(|e| ExecError { error: e, origin: ErrorOrigin::Callee })?;
1301
1302			Ok(output)
1303		};
1304
1305		// All changes performed by the contract are executed under a storage transaction.
1306		// This allows for roll back on error. Changes to the cached contract_info are
1307		// committed or rolled back when popping the frame.
1308		//
1309		// `with_transactional` may return an error caused by a limit in the
1310		// transactional storage depth.
1311		let transaction_outcome =
1312			with_transaction(|| -> TransactionOutcome<Result<_, DispatchError>> {
1313				let output = do_transaction();
1314				match &output {
1315					Ok(result) if !result.did_revert() =>
1316						TransactionOutcome::Commit(Ok((true, output))),
1317					_ => TransactionOutcome::Rollback(Ok((false, output))),
1318				}
1319			});
1320
1321		let (success, output) = match transaction_outcome {
1322			// `with_transactional` executed successfully, and we have the expected output.
1323			Ok((success, output)) => {
1324				if_tracing(|tracer| {
1325					let gas_consumed = top_frame!(self).nested_gas.gas_consumed();
1326					match &output {
1327						Ok(output) => tracer.exit_child_span(&output, gas_consumed),
1328						Err(e) => tracer.exit_child_span_with_error(e.error.into(), gas_consumed),
1329					}
1330				});
1331
1332				(success, output)
1333			},
1334			// `with_transactional` returned an error, and we propagate that error and note no state
1335			// has changed.
1336			Err(error) => {
1337				if_tracing(|tracer| {
1338					let gas_consumed = top_frame!(self).nested_gas.gas_consumed();
1339					tracer.exit_child_span_with_error(error.into(), gas_consumed);
1340				});
1341
1342				(false, Err(error.into()))
1343			},
1344		};
1345
1346		if success {
1347			self.transient_storage.commit_transaction();
1348		} else {
1349			self.transient_storage.rollback_transaction();
1350		}
1351
1352		log::trace!(target: LOG_TARGET, "frame finished with: {output:?}");
1353
1354		self.pop_frame(success);
1355		output.map(|output| {
1356			self.top_frame_mut().last_frame_output = output;
1357		})
1358	}

1360	/// Remove the current (top) frame from the stack.
1361	///
1362	/// This is called after running the current frame. It commits cached values to storage
1363	/// and invalidates all stale references to it that might exist further down the call stack.
1364	fn pop_frame(&mut self, persist: bool) {
1365		// Pop the current frame from the stack and return it in case it needs to interact
1366		// with duplicates that might exist on the stack.
1367		// A `None` means that we are returning from the `first_frame`.
1368		let frame = self.frames.pop();
1369
1370		// Both branches do essentially the same with the exception. The difference is that
1371		// the else branch does consume the hardcoded `first_frame`.
1372		if let Some(mut frame) = frame {
1373			let account_id = &frame.account_id;
1374			let prev = top_frame_mut!(self);
1375
1376			prev.nested_gas.absorb_nested(frame.nested_gas);
1377
1378			// Only gas counter changes are persisted in case of a failure.
1379			if !persist {
1380				return;
1381			}
1382
1383			// Record the storage meter changes of the nested call into the parent meter.
1384			// If the dropped frame's contract has a contract info we update the deposit
1385			// counter in its contract info. The load is necessary to pull it from storage in case
1386			// it was invalidated.
1387			frame.contract_info.load(account_id);
1388			let mut contract = frame.contract_info.into_contract();
1389			prev.nested_storage.absorb(frame.nested_storage, account_id, contract.as_mut());
1390
1391			// In case the contract wasn't terminated we need to persist changes made to it.
1392			if let Some(contract) = contract {
1393				// optimization: Predecessor is the same contract.
1394				// We can just copy the contract into the predecessor without a storage write.
1395				// This is possible when there is no other contract in-between that could
1396				// trigger a rollback.
1397				if prev.account_id == *account_id {
1398					prev.contract_info = CachedContract::Cached(contract);
1399					return;
1400				}
1401
1402				// Predecessor is a different contract: We persist the info and invalidate the first
1403				// stale cache we find. This triggers a reload from storage on next use. We skip(1)
1404				// because that case is already handled by the optimization above. Only the first
1405				// cache needs to be invalidated because that one will invalidate the next cache
1406				// when it is popped from the stack.
1407				AccountInfo::<T>::insert_contract(
1408					&T::AddressMapper::to_address(account_id),
1409					contract,
1410				);
1411				if let Some(f) = self.frames_mut().skip(1).find(|f| f.account_id == *account_id) {
1412					f.contract_info.invalidate();
1413				}
1414			}
1415		} else {
1416			// TODO: iterate contracts_to_be_destroyed and destroy each contract
1417			self.gas_meter.absorb_nested(mem::take(&mut self.first_frame.nested_gas));
1418			if !persist {
1419				return;
1420			}
1421			let mut contract = self.first_frame.contract_info.as_contract();
1422			self.storage_meter.absorb(
1423				mem::take(&mut self.first_frame.nested_storage),
1424				&self.first_frame.account_id,
1425				contract.as_deref_mut(),
1426			);
1427			if let Some(contract) = contract {
1428				AccountInfo::<T>::insert_contract(
1429					&T::AddressMapper::to_address(&self.first_frame.account_id),
1430					contract.clone(),
1431				);
1432			}
1433		}
1434	}

1436	/// Transfer some funds from `from` to `to`.
1437	///
1438	/// This is a no-op for zero `value`, avoiding events to be emitted for zero balance transfers.
1439	///
1440	/// If the destination account does not exist, it is pulled into existence by transferring the
1441	/// ED from `origin` to the new account. The total amount transferred to `to` will be ED +
1442	/// `value`. This makes the ED fully transparent for contracts.
1443	/// The ED transfer is executed atomically with the actual transfer, avoiding the possibility of
1444	/// the ED transfer succeeding but the actual transfer failing. In other words, if the `to` does
1445	/// not exist, the transfer does fail and nothing will be sent to `to` if either `origin` can
1446	/// not provide the ED or transferring `value` from `from` to `to` fails.
1447	/// Note: This will also fail if `origin` is root.
1448	fn transfer<S: storage::meter::State + Default + Debug>(
1449		origin: &Origin<T>,
1450		from: &T::AccountId,
1451		to: &T::AccountId,
1452		value: U256,
1453		storage_meter: &mut storage::meter::GenericMeter<T, S>,
1454	) -> DispatchResult {
1455		fn transfer_with_dust<T: Config>(
1456			from: &AccountIdOf<T>,
1457			to: &AccountIdOf<T>,
1458			value: BalanceWithDust<BalanceOf<T>>,
1459		) -> DispatchResult {
1460			let (value, dust) = value.deconstruct();
1461
1462			fn transfer_balance<T: Config>(
1463				from: &AccountIdOf<T>,
1464				to: &AccountIdOf<T>,
1465				value: BalanceOf<T>,
1466			) -> DispatchResult {
1467				T::Currency::transfer(from, to, value, Preservation::Preserve)
1468				.map_err(|err| {
1469					log::debug!(target: crate::LOG_TARGET, "Transfer failed: from {from:?} to {to:?} (value: ${value:?}). Err: {err:?}");
1470					Error::<T>::TransferFailed
1471				})?;
1472				Ok(())
1473			}
1474
1475			fn transfer_dust<T: Config>(
1476				from: &mut AccountInfo<T>,
1477				to: &mut AccountInfo<T>,
1478				dust: u32,
1479			) -> DispatchResult {
1480				from.dust =
1481					from.dust.checked_sub(dust).ok_or_else(|| Error::<T>::TransferFailed)?;
1482				to.dust = to.dust.checked_add(dust).ok_or_else(|| Error::<T>::TransferFailed)?;
1483				Ok(())
1484			}
1485
1486			if dust.is_zero() {
1487				return transfer_balance::<T>(from, to, value)
1488			}
1489
1490			let from_addr = <T::AddressMapper as AddressMapper<T>>::to_address(from);
1491			let mut from_info = AccountInfoOf::<T>::get(&from_addr).unwrap_or_default();
1492
1493			let to_addr = <T::AddressMapper as AddressMapper<T>>::to_address(to);
1494			let mut to_info = AccountInfoOf::<T>::get(&to_addr).unwrap_or_default();
1495
1496			let plank = T::NativeToEthRatio::get();
1497
1498			if from_info.dust < dust {
1499				T::Currency::burn_from(
1500					from,
1501					1u32.into(),
1502					Preservation::Preserve,
1503					Precision::Exact,
1504					Fortitude::Polite,
1505				)
1506				.map_err(|err| {
1507					log::debug!(target: crate::LOG_TARGET, "Burning 1 plank from {from:?} failed. Err: {err:?}");
1508					Error::<T>::TransferFailed
1509				})?;
1510
1511				from_info.dust =
1512					from_info.dust.checked_add(plank).ok_or_else(|| Error::<T>::TransferFailed)?;
1513			}
1514
1515			transfer_balance::<T>(from, to, value)?;
1516			transfer_dust::<T>(&mut from_info, &mut to_info, dust)?;
1517
1518			if to_info.dust >= plank {
1519				T::Currency::mint_into(to, 1u32.into())?;
1520				to_info.dust =
1521					to_info.dust.checked_sub(plank).ok_or_else(|| Error::<T>::TransferFailed)?;
1522			}
1523
1524			AccountInfoOf::<T>::set(&from_addr, Some(from_info));
1525			AccountInfoOf::<T>::set(&to_addr, Some(to_info));
1526
1527			Ok(())
1528		}
1529
1530		let value = BalanceWithDust::<BalanceOf<T>>::from_value::<T>(value)?;
1531		if value.is_zero() {
1532			return Ok(());
1533		}
1534
1535		if <System<T>>::account_exists(to) {
1536			return transfer_with_dust::<T>(from, to, value)
1537		}
1538
1539		let origin = origin.account_id()?;
1540		let ed = <T as Config>::Currency::minimum_balance();
1541		with_transaction(|| -> TransactionOutcome<DispatchResult> {
1542			match storage_meter
1543				.record_charge(&StorageDeposit::Charge(ed))
1544				.and_then(|_| {
1545					T::Currency::transfer(origin, to, ed, Preservation::Preserve)
1546						.map_err(|_| Error::<T>::StorageDepositNotEnoughFunds.into())
1547				})
1548				.and_then(|_| transfer_with_dust::<T>(from, to, value))
1549			{
1550				Ok(_) => TransactionOutcome::Commit(Ok(())),
1551				Err(err) => TransactionOutcome::Rollback(Err(err)),
1552			}
1553		})
1554	}

1856impl<'a, T, E> PrecompileExt for Stack<'a, T, E>
1857where
1858	T: Config,
1859	E: Executable<T>,
1860	BalanceOf<T>: Into<U256> + TryFrom<U256>,
1861	MomentOf<T>: Into<U256>,
1862	T::Hash: frame_support::traits::IsType<H256>,
1863{

1866	fn call(
1867		&mut self,
1868		gas_limit: Weight,
1869		deposit_limit: U256,
1870		dest_addr: &H160,
1871		value: U256,
1872		input_data: Vec<u8>,
1873		allows_reentry: bool,
1874		read_only: bool,
1875	) -> Result<(), ExecError> {
1876		// Before pushing the new frame: Protect the caller contract against reentrancy attacks.
1877		// It is important to do this before calling `allows_reentry` so that a direct recursion
1878		// is caught by it.
1879		self.top_frame_mut().allows_reentry = allows_reentry;
1880
1881		// We reset the return data now, so it is cleared out even if no new frame was executed.
1882		// This is for example the case for balance transfers or when creating the frame fails.
1883		*self.last_frame_output_mut() = Default::default();
1884
1885		let try_call = || {
1886			// Enable read-only access if requested; cannot disable it if already set.
1887			let is_read_only = read_only || self.is_read_only();
1888
1889			// We can skip the stateful lookup for pre-compiles.
1890			let dest = if <AllPrecompiles<T>>::get::<Self>(dest_addr.as_fixed_bytes()).is_some() {
1891				T::AddressMapper::to_fallback_account_id(dest_addr)
1892			} else {
1893				T::AddressMapper::to_account_id(dest_addr)
1894			};
1895
1896			if !self.allows_reentry(&dest) {
1897				return Err(<Error<T>>::ReentranceDenied.into());
1898			}
1899
1900			// We ignore instantiate frames in our search for a cached contract.
1901			// Otherwise it would be possible to recursively call a contract from its own
1902			// constructor: We disallow calling not fully constructed contracts.
1903			let cached_info = self
1904				.frames()
1905				.find(|f| f.entry_point == ExportedFunction::Call && f.account_id == dest)
1906				.and_then(|f| match &f.contract_info {
1907					CachedContract::Cached(contract) => Some(contract.clone()),
1908					_ => None,
1909				});
1910
1911			if let Some(executable) = self.push_frame(
1912				FrameArgs::Call { dest: dest.clone(), cached_info, delegated_call: None },
1913				value,
1914				gas_limit,
1915				deposit_limit.saturated_into::<BalanceOf<T>>(),
1916				is_read_only,
1917			)? {
1918				self.run(executable, input_data, BumpNonce::Yes)
1919			} else {
1920				if_tracing(|t| {
1921					t.enter_child_span(
1922						T::AddressMapper::to_address(self.account_id()),
1923						T::AddressMapper::to_address(&dest),
1924						false,
1925						is_read_only,
1926						value,
1927						&input_data,
1928						Weight::zero(),
1929					);
1930				});
1931
1932				let result = if is_read_only && value.is_zero() {
1933					Ok(Default::default())
1934				} else if is_read_only {
1935					Err(Error::<T>::StateChangeDenied.into())
1936				} else {
1937					let account_id = self.account_id().clone();
1938					let frame = top_frame_mut!(self);
1939					Self::transfer_from_origin(
1940						&self.origin,
1941						&Origin::from_account_id(account_id),
1942						&dest,
1943						value,
1944						&mut frame.nested_storage,
1945					)
1946				};
1947
1948				if_tracing(|t| match result {
1949					Ok(ref output) => t.exit_child_span(&output, Weight::zero()),
1950					Err(e) => t.exit_child_span_with_error(e.error.into(), Weight::zero()),
1951				});
1952
1953				result.map(|_| ())
1954			}
1955		};
1956
1957		// We need to make sure to reset `allows_reentry` even on failure.
1958		let result = try_call();
1959
1960		// Protection is on a per call basis.
1961		self.top_frame_mut().allows_reentry = true;
1962
1963		result
1964	}
