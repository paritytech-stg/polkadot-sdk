63impl<T: Config> ContractBlob<T> {
64	/// Create a new contract from EVM init code.
65	pub fn from_evm_init_code(code: Vec<u8>, owner: AccountIdOf<T>) -> Result<Self, DispatchError> {
66		if code.len() > revm::primitives::eip3860::MAX_INITCODE_SIZE {
67			return Err(<Error<T>>::BlobTooLarge.into());
68		}
69
70		// EIP-3541: Reject new contract code starting with the 0xEF byte
71		if code.first() == Some(&0xEF) {
72			return Err(<Error<T>>::CodeRejected.into());
73		}
74
75		let code_len = code.len() as u32;
76		let code_info = CodeInfo {
77			owner,
78			deposit: Default::default(),
79			refcount: 0,
80			code_len,
81			code_type: BytecodeType::Evm,
82			behaviour_version: Default::default(),
83		};
84
85		Bytecode::new_raw_checked(Bytes::from(code.to_vec())).map_err(|err| {
86			log::debug!(target: LOG_TARGET, "failed to create evm bytecode from init code: {err:?}" );
87			<Error<T>>::CodeRejected
88		})?;
89
90		// Code hash is not relevant for init code, since it is not stored on-chain.
91		let code_hash = H256::default();
92		Ok(ContractBlob { code, code_info, code_hash })
93	}
94
95	/// Create a new contract from EVM runtime code.
96	pub fn from_evm_runtime_code(
97		code: Vec<u8>,
98		owner: AccountIdOf<T>,
99	) -> Result<Self, DispatchError> {
100		if code.len() > revm::primitives::eip170::MAX_CODE_SIZE {
101			return Err(<Error<T>>::BlobTooLarge.into());
102		}
103
104		let code_len = code.len() as u32;
105		let deposit = super::calculate_code_deposit::<T>(code_len);
106
107		let code_info = CodeInfo {
108			owner,
109			deposit,
110			refcount: 0,
111			code_len,
112			code_type: BytecodeType::Evm,
113			behaviour_version: Default::default(),
114		};
115
116		Bytecode::new_raw_checked(Bytes::from(code.to_vec())).map_err(|err| {
117			log::debug!(target: LOG_TARGET, "failed to create evm bytecode from code: {err:?}" );
118			<Error<T>>::CodeRejected
119		})?;
120
121		let code_hash = H256(sp_io::hashing::keccak_256(&code));
122		Ok(ContractBlob { code, code_info, code_hash })
123	}
124}

126/// Calls the EVM interpreter with the provided bytecode and inputs.
127pub fn call<'a, E: Ext>(bytecode: Bytecode, ext: &'a mut E, inputs: EVMInputs) -> ExecResult {
128	let mut interpreter: Interpreter<EVMInterpreter<'a, E>> = Interpreter {
129		gas: Gas::default(),
130		bytecode: ExtBytecode::new(bytecode),
131		stack: Stack::new(),
132		return_data: Default::default(),
133		memory: SharedMemory::new(),
134		input: inputs,
135		runtime_flag: RuntimeFlags { is_static: ext.is_read_only(), spec_id: SpecId::default() },
136		extend: ext,
137	};
138
139	let table = instruction_table::<'a, E>();
140	let result = run(&mut interpreter, &table);
141
142	instruction_result_into_exec_error::<E>(result.result)
143		.map(Err)
144		.unwrap_or_else(|| {
145			Ok(ExecReturnValue {
146				flags: if result.is_revert() { ReturnFlags::REVERT } else { ReturnFlags::empty() },
147				data: result.output.to_vec(),
148			})
149		})
150}

152/// Runs the EVM interpreter
153fn run<'a, E: Ext>(
154	interpreter: &mut Interpreter<EVMInterpreter<'a, E>>,
155	table: &revm::interpreter::InstructionTable<EVMInterpreter<'a, E>, DummyHost>,
156) -> InterpreterResult {
157	let host = &mut DummyHost {};
158	loop {
159		let action = interpreter.run_plain(table, host);
160		match action {
161			InterpreterAction::Return(result) => {
162				log::trace!(target: LOG_TARGET, "Evm return {:?}", result);
163				debug_assert!(
164					result.gas == Default::default(),
165					"Interpreter gas state is unused; found: {:?}",
166					result.gas,
167				);
168				return result;
169			},
170			InterpreterAction::NewFrame(frame_input) => match frame_input {
171				FrameInput::Call(call_input) => run_call(interpreter, call_input),
172				FrameInput::Create(create_input) => run_create(interpreter, create_input),
173				FrameInput::Empty => unreachable!(),
174			},
175		}
176	}
177}

179fn run_call<'a, E: Ext>(
180	interpreter: &mut Interpreter<EVMInterpreter<'a, E>>,
181	call_input: Box<CallInputs>,
182) {
183	let callee: H160 = if call_input.scheme.is_delegate_call() {
184		call_input.bytecode_address.0 .0.into()
185	} else {
186		call_input.target_address.0 .0.into()
187	};
188
189	let input = match &call_input.input {
190		CallInput::Bytes(bytes) => bytes.to_vec(),
191		CallInput::SharedBuffer(range) => interpreter.memory.global_slice(range.clone()).to_vec(),
192	};
193	let call_result = match call_input.scheme {
194		CallScheme::Call | CallScheme::StaticCall => interpreter.extend.call(
195			Weight::from_parts(call_input.gas_limit, u64::MAX),
196			U256::MAX,
197			&callee,
198			U256::from_revm_u256(&call_input.call_value()),
199			input,
200			true,
201			call_input.is_static,
202		),
203		CallScheme::CallCode => {
204			unreachable!()
205		},
206		CallScheme::DelegateCall => interpreter.extend.delegate_call(
207			Weight::from_parts(call_input.gas_limit, u64::MAX),
208			U256::MAX,
209			callee,
210			input,
211		),
212	};
213
214	let (return_data, did_revert) = {
215		let return_value = interpreter.extend.last_frame_output();
216		let return_data: Bytes = return_value.data.clone().into();
217		(return_data, return_value.did_revert())
218	};
219
220	let mem_length = call_input.return_memory_offset.len();
221	let mem_start = call_input.return_memory_offset.start;
222	let returned_len = return_data.len();
223	let target_len = min(mem_length, returned_len);
224
225	interpreter.return_data.set_buffer(return_data);
226
227	match call_result {
228		Ok(()) => {
229			// success or revert
230			gas!(interpreter, RuntimeCosts::CopyToContract(target_len as u32));
231			interpreter
232				.memory
233				.set(mem_start, &interpreter.return_data.buffer()[..target_len]);
234			let _ = interpreter.stack.push(primitives::U256::from(!did_revert as u8));
235		},
236		Err(err) => {
237			let _ = interpreter.stack.push(primitives::U256::ZERO);
238			if let Some(reason) = exec_error_into_halt_reason::<E>(err) {
239				interpreter.halt(reason);
240			}
241		},
242	}
243}

245fn run_create<'a, E: Ext>(
246	interpreter: &mut Interpreter<EVMInterpreter<'a, E>>,
247	create_input: Box<CreateInputs>,
248) {
249	let value = U256::from_revm_u256(&create_input.value);
250
251	let salt = match create_input.scheme {
252		CreateScheme::Create => None,
253		CreateScheme::Create2 { salt } => Some(salt.to_le_bytes()),
254		CreateScheme::Custom { .. } => unreachable!("custom create schemes are not supported"),
255	};
256
257	let call_result = interpreter.extend.instantiate(
258		Weight::from_parts(create_input.gas_limit, u64::MAX),
259		U256::MAX,
260		Code::Upload(create_input.init_code.to_vec()),
261		value,
262		vec![],
263		salt.as_ref(),
264	);
265
266	let return_value = interpreter.extend.last_frame_output();
267	let return_data: Bytes = return_value.data.clone().into();
268
269	match call_result {
270		Ok(address) => {
271			if return_value.did_revert() {
272				// Contract creation reverted โ return data must be propagated
273				gas!(interpreter, RuntimeCosts::CopyToContract(return_data.len() as u32));
274				interpreter.return_data.set_buffer(return_data);
275				let _ = interpreter.stack.push(primitives::U256::ZERO);
276			} else {
277				// Otherwise clear it. Note that RETURN opcode should abort.
278				interpreter.return_data.clear();
279				let stack_item: Address = address.0.into();
280				let _ = interpreter.stack.push(stack_item.into_word().into());
281			}
282		},
283		Err(err) => {
284			let _ = interpreter.stack.push(primitives::U256::ZERO);
285			if let Some(reason) = exec_error_into_halt_reason::<E>(err) {
286				interpreter.halt(reason);
287			}
288		},
289	}
290}

352/// Conversion of a `ExecError` to `ReturnErrorCode`.
353///
354/// Used when converting the error returned from a subcall in order to map it to the
355/// equivalent EVM interpreter [InstructionResult].
356///
357/// - Returns `None` when the caller can recover the error.
358/// - Otherwise, some [InstructionResult] error code (the halt reason) is returned. Most [ExecError]
359///   variants don't map to a [InstructionResult]. The conversion is lossy and defaults to
360///   [InstructionResult::Revert] for most cases.
361///
362/// Uses the overarching [super::exec_error_into_return_code] method to determine if
363/// the error is recoverable or not. This guarantees consistent behavior accross both
364/// VM backends.
365fn exec_error_into_halt_reason<E: Ext>(from: ExecError) -> Option<InstructionResult> {
366	log::trace!("call frame execution error in EVM caller: {:?}", &from);
367
368	if super::exec_error_into_return_code::<E>(from).is_ok() {
369		return None;
370	}
371
372	let static_memory_too_large = Error::<E::T>::StaticMemoryTooLarge.into();
373	let code_rejected = Error::<E::T>::CodeRejected.into();
374	let transfer_failed = Error::<E::T>::TransferFailed.into();
375	let duplicate_contract = Error::<E::T>::DuplicateContract.into();
376	let balance_conversion_failed = Error::<E::T>::BalanceConversionFailed.into();
377	let value_too_large = Error::<E::T>::ValueTooLarge.into();
378	let out_of_gas = Error::<E::T>::OutOfGas.into();
379	let out_of_deposit = Error::<E::T>::StorageDepositLimitExhausted.into();
380
381	Some(match from.error {
382		err if err == static_memory_too_large => InstructionResult::MemoryLimitOOG,
383		err if err == code_rejected => InstructionResult::OpcodeNotFound,
384		err if err == transfer_failed => InstructionResult::OutOfFunds,
385		err if err == duplicate_contract => InstructionResult::CreateCollision,
386		err if err == balance_conversion_failed => InstructionResult::OverflowPayment,
387		err if err == value_too_large => InstructionResult::OverflowPayment,
388		err if err == out_of_deposit => InstructionResult::OutOfFunds,
389		err if err == out_of_gas => InstructionResult::OutOfGas,
390		_ => InstructionResult::Revert,
391	})
392}

394/// Map [InstructionResult] into an [ExecError] for passing it up the stack.
395///
396/// Returns `None` if the instruction result is not an error case.
397fn instruction_result_into_exec_error<E: Ext>(from: InstructionResult) -> Option<ExecError> {
398	match from {
399		InstructionResult::OutOfGas |
400		InstructionResult::InvalidOperandOOG |
401		InstructionResult::ReentrancySentryOOG |
402		InstructionResult::PrecompileOOG |
403		InstructionResult::MemoryOOG => Some(Error::<E::T>::OutOfGas),
404		InstructionResult::MemoryLimitOOG => Some(Error::<E::T>::StaticMemoryTooLarge),
405		InstructionResult::OpcodeNotFound |
406		InstructionResult::InvalidJump |
407		InstructionResult::NotActivated |
408		InstructionResult::InvalidFEOpcode |
409		InstructionResult::CreateContractStartingWithEF => Some(Error::<E::T>::InvalidInstruction),
410		InstructionResult::CallNotAllowedInsideStatic |
411		InstructionResult::StateChangeDuringStaticCall => Some(Error::<E::T>::StateChangeDenied),
412		InstructionResult::StackUnderflow |
413		InstructionResult::StackOverflow |
414		InstructionResult::NonceOverflow |
415		InstructionResult::PrecompileError |
416		InstructionResult::FatalExternalError => Some(Error::<E::T>::ContractTrapped),
417		InstructionResult::OutOfOffset => Some(Error::<E::T>::OutOfBounds),
418		InstructionResult::CreateCollision => Some(Error::<E::T>::DuplicateContract),
419		InstructionResult::OverflowPayment => Some(Error::<E::T>::BalanceConversionFailed),
420		InstructionResult::CreateContractSizeLimit | InstructionResult::CreateInitCodeSizeLimit =>
421			Some(Error::<E::T>::StaticMemoryTooLarge),
422		InstructionResult::CallTooDeep => Some(Error::<E::T>::MaxCallDepthReached),
423		InstructionResult::OutOfFunds => Some(Error::<E::T>::TransferFailed),
424		InstructionResult::CreateInitCodeStartingEF00 |
425		InstructionResult::InvalidEOFInitCode |
426		InstructionResult::InvalidExtDelegateCallTarget => Some(Error::<E::T>::ContractTrapped),
427		InstructionResult::Stop |
428		InstructionResult::Return |
429		InstructionResult::Revert |
430		InstructionResult::SelfDestruct => None,
431	}
432	.map(Into::into)
433}

435/// Blanket conversion trait between `sp_core::U256` and `revm::primitives::U256`
436pub trait U256Converter {
437	/// Convert `self` into `revm::primitives::U256`
438	fn into_revm_u256(&self) -> revm::primitives::U256;
439
440	/// Convert from `revm::primitives::U256` into `Self`
441	fn from_revm_u256(value: &revm::primitives::U256) -> Self;
442}
443
444impl U256Converter for sp_core::U256 {
445	fn into_revm_u256(&self) -> revm::primitives::U256 {
446		let bytes = self.to_big_endian();
447		revm::primitives::U256::from_be_bytes(bytes)
448	}
449
450	fn from_revm_u256(value: &revm::primitives::U256) -> Self {
451		let bytes = value.to_be_bytes::<32>();
452		sp_core::U256::from_big_endian(&bytes)
453	}
